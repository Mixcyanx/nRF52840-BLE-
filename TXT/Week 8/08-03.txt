import java.util.*;
public class main {
public static void main(String[] args) {
Scanner in = new Scanner(System.in);
String d=in.next();
System.out.println(d);
System.out.println(INSERT(d,3,'A'));
System.out.println(ERASE(d,8)); }
static String INSERT(String d,int i,char c) {
return d.substring(0,i)+c+d.substring(i); }
static String ERASE(String d,int i) {
return d.substring(0,i)+d.substring(i+1); }}
