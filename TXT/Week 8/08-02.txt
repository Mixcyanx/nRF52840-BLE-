import java.util.*;
public class main {
public static void main(String[] args) {
Scanner in = new Scanner(System.in);
String d = in.next();
System.out.println(d);
System.out.println(PUSH_FRONT(d,'A'));
System.out.println(PUSH_BACK(d,'Z')); }
static String PUSH_FRONT(String d,char c) {
return c + d; }
static String PUSH_BACK(String d,char c) {
return d + c; }}
