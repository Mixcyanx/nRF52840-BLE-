import java.util.*;
public class main {
public static void main(String[] args) {
Scanner in=new Scanner(System.in);
String s=in.next();
boolean ok=true;
for(int i=0;i<s.length()/2;i++){
if(s.charAt(i)!=s.charAt(s.length()-1-i)){ok=false;break;}}
if(ok)System.out.print("Yes\r\n");
else System.out.print("No\r\n");}}
