import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
int n = s.nextInt();
System.out.println("f(" + n + ")=" + f(n)); }
static int f(int n) {
return n == 0 ? 1 : sum(n, 0); }
static int sum(int n, int i) {
return i == n ? 0 : f(i) * f(n - 1 - i) + sum(n, i + 1); }}
