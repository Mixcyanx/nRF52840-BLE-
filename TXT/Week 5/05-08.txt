import java.util.*;
public class main {
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
int n = s.nextInt();
f(n, 'A', 'B', 'C'); }
static void f(int n, char a, char b, char c) {
if (n == 1) System.out.println(a + "-->" + c);
else {
f(n - 1, a, c, b);
System.out.println(a + "-->" + c);
f(n - 1, b, a, c); }}}
