import java.util.*;
public class main {
static double f(int n) {
if (n == 1) return 1.0;
return (double)n / (n + 1) * f(n - 1); }
public static void main(String[] args) {
Scanner s = new Scanner(System.in);
if (s.hasNextInt()) {
int n = s.nextInt();
double res = f(n);
System.out.print("a(" + n + ")=" + res); }}}
